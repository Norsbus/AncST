# AncST Downstream Analysis Requirements
# Updated: February 2026
#
# RECOMMENDED INSTALLATION METHOD:
#   Use the install_downstream_requirements.sh script instead of pip install
#   It will check for existing installations and handle conda+pip packages properly:
#
#   ./install_downstream_requirements.sh
#
# This requirements.txt is kept for reference and manual pip installations only.
#
# PYTHON VERSION: 3.12+ (all packages compatible as of Feb 2026)
# Older versions: 3.11, 3.10, 3.9 also work with adjusted package versions

# CORE SCIENTIFIC COMPUTING
# These should be installed via conda for best performance (optimized BLAS/LAPACK)
numpy>=2.0,<3.0          # Array operations, linear algebra
scipy>=1.14,<2.0         # Scientific computing, sparse matrices
biopython>=1.84          # Sequence analysis, phylogenetics
matplotlib>=3.8,<4.0     # Plotting and visualization
pandas>=2.2,<4.0         # Data tables and analysis

# NETWORK AND GRAPH ANALYSIS
networkx>=3.2            # Graph algorithms (cographs, connected components)
markov-clustering==0.0.6.dev0  # MCL algorithm for clustering (pip only)

# SPECIALIZED PACKAGES
dbscan1d>=0.2.0         # 1D DBSCAN clustering (pip only)
ete3>=3.1.3             # Phylogenetic tree manipulation
tralda>=0.2.0           # Tree data structures, cograph algorithms (pip only)
pygenomeviz>=1.4.0      # Genome visualization for comparative genomics

# BIOINFORMATICS TOOLS (install via conda from bioconda channel)
# These are binary tools, not pip packages. Install with:
#   conda install -c bioconda <tool-name>
#
# orthofinder>=3.0    # Phylogenetic orthology inference (v3.x: 5-7% more accurate!)
#   Version 3.x improvements: --assign option, ASTRAL-Pro3, reduced RAM/runtime
#   Dependencies (auto-installed with orthofinder):
#   - mcl             # MCL clustering algorithm
#   - diamond=2.0.15  # Sequence search (NOTE: use 2.0.15, issues with 2.1+)
#   - fastme          # Fast distance-based phylogeny
#
# miniprot           # Protein-to-genome alignment (faster than tblastn)

# INSTALLATION NOTES
#
# 1. CONDA VS PIP:
#    - Use conda for: numpy, scipy, biopython, matplotlib, pandas, networkx,
#                     ete3, pygenomeviz, orthofinder, miniprot
#    - Use pip for: markov-clustering, dbscan1d, tralda
#
# 2. PYTHON 3.12+ COMPATIBILITY:
#    All packages have been verified to work with Python 3.12 as of Feb 2026.
#    Older requirements.txt mentioned Python 3.11 restriction - this is outdated.
#
# 3. ARCHITECTURE SUPPORT:
#    - Linux x86_64 (Intel/AMD): supported
#    - Linux ARM64 (aarch64):    supported
#    - macOS x86_64 (Intel):     supported
#    - macOS ARM64 (M1/M2/M3):   supported
#    - 32-bit systems:           not supported
#
# 4. KNOWN ISSUES:
#    - markov-clustering: Last updated 2018, but still works. No conda package.
#    - DIAMOND: Versions 2.1+ have compatibility issues with OrthoFinder.
#               Use diamond=2.0.15 specifically.
#
# 5. MANUAL INSTALLATION (if script fails):
#
#    conda create -n AncST_downstream python=3.12
#    conda activate AncST_downstream
#
#    # Core packages via conda
#    conda install -c conda-forge -c bioconda \
#        numpy scipy biopython matplotlib pandas networkx \
#        ete3 pygenomeviz
#
#    # Pip-only packages
#    pip install markov-clustering==0.0.6.dev0 dbscan1d tralda
#
#    # Tools via conda
#    conda install -c bioconda 'orthofinder>=3.0' diamond=2.0.15 mcl fastme miniprot
